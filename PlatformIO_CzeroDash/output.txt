Battery Pack Cell voltages and Module Temperatures

---------
6E1, 6E2, 6E3, and 6E4 are a secquence of messages that
contains cell voltages (88 voltages) and teperatures (66 of them).

Presumably, there are 22 modules, each with 4 cells and 3 temperatures.

----------
6E1 is immediately followed by 6E2, then a about 20 ms later,
there is a 6E3, imediately followed by a 6E4. This group of 4
messages all have the same Index (1 through 12), and 12 such
groups, each with an increasing Index, make a full Set of
messages that carry data for 88 voltages and 66 temperatures.

The Set repeats about twice a second, I believe.

All of these messages use D1 as the Index (01 to 0C), and they
all use bytes D5 & D6 for one voltage, and D7 & D8 as the other
voltage. Each voltage is multiplied by 100 and stored as a 2-byte
integer. Values typically range from about 250 to 400 (2.5 volts
to 4.00 volts). The 4 messags, with 12 Index values, and two
voltages in each one ... have space for 96 voltages, but they
actually contain only data representing 88 voltages.

Similarly, there is space for 4 * 12 * 4 temperatures, but there
are only 2 temps in most of these messages, and even then,
there are only 66 temperatures used.

The unused data is typically whole bytes of zero value, I think.

6D1:D1 is the Index, value 01 to 0C
6D1:D2 is zero
6D1:D3 and D4 are two one-byte temperatures
6D1:D5 & D6 as ((D5 * 256) + D6) / 100 = a cell voltage
6D1:D7 & D8 as ((D7 * 256) + D8) / 100 = a cell voltage

6D2:D1 is the Index, value 01 to 0C
6D2:D2 and D3 are two one-byte temperatures
6D2:D4 is zero
6D2:D5 & D6 as ((D5 * 256) + D6) / 100 = a cell voltage
6D2:D7 & D8 as ((D7 * 256) + D8) / 100 = a cell voltage

6D3:D1 is the Index, value 01 to 0C
6D3:D2 and D3 are two one-byte temperatures
6D3:D4 is zero
6D3:D5 & D6 as ((D5 * 256) + D6) / 100 = a cell voltage
6D3:D7 & D8 as ((D7 * 256) + D8) / 100 = a cell voltage

6D4:D1 is the Index, value 01 to 0C
6D4:D2, D3, and D4 are zero (no temperatures)
6D4:D5 & D6 as ((D5 * 256) + D6) / 100 = a cell voltage
6D4:D7 & D8 as ((D7 * 256) + D8) / 100 = a cell voltage

Note that not all of the space for voltages or temperatures
is actually used.

      cellTemperatures[index*6 +0 +indexOffsetT] = data[2] -50;
      cellTemperatures[index*6 +1 +indexOffsetT] = data[3] -50;
      cellVoltages[index*8 +0 +indexOffsetV] = data[4]*256 + data[5];
      cellVoltages[index*8 +1 +indexOffsetV] = data[6]*256 + data[7];
    }

    if (CAN.packetId() == 0x6E2) {
      CAN.readBytes(data, 8);
      index = data[0]-1;
      if (index > 5) {
        indexOffsetT = -2;
        indexOffsetV = -4;
      }
      cellTemperatures[index*6 +2 +indexOffsetT] = data[1] -50;
      cellTemperatures[index*6 +3 +indexOffsetT] = data[2] -50;
      cellVoltages[index*8 +2 +indexOffsetV] = data[4]*256 + data[5];
      cellVoltages[index*8 +3 +indexOffsetV] = data[6]*256 + data[7];
    }

    if (CAN.packetId() == 0x6E3) {
      CAN.readBytes(data, 8);
      index = data[0]-1;
      if (index > 5) {
        indexOffsetT = -2;
        indexOffsetV = -4;
      }
      if (index == 5 || index == 11) return;
      cellTemperatures[index*6 +4 +indexOffsetT] = data[1] -50;
      cellTemperatures[index*6 +5 +indexOffsetT] = data[2] -50;
      cellVoltages[index*8 +4 +indexOffsetV] = data[4]*256 + data[5];
      cellVoltages[index*8 +5 +indexOffsetV] = data[6]*256 + data[7];
    }

    if (CAN.packetId() == 0x6E4) {
      CAN.readBytes(data, 8);
      index = data[0]-1;
      if (index > 5) {
        indexOffsetV = -4;
      }
      if (index == 5 || index == 11) return;
      cellVoltages[index*8 +6 +indexOffsetV] = data[4]*256 + data[5];
      cellVoltages[index*8 +7 +indexOffsetV] = data[6]*256 + data[7];
    }

    if (CAN.packetId() == 0x373) {
      CAN.readBytes(data, 8);
      cellMaxVoltage = (data[0] + 210) / 100.0;
      cellMinVoltage = (data[1] + 210) / 100.0;
      battAmp = (((data[2] * 256) + data[3]) - 32768.0)* -0.01;
      battVolt = ((data[4] * 256) + data[5])* 0.1;
      powerKW = (battAmp*battVolt)/1000.0;
      powerSum += (powerKW / 30.0);
      powerNum++;
    }

    if (CAN.packetId() == 0x374) {
      CAN.readBytes(data, 8);
      soc1 = (data[0] - 10.0) / 2.0;
      soc2 = (data[1] - 10.0) / 2.0;
    }
  }
}

void setup() {
  // start serial
  Serial.begin(9600);
  delay(1000);

  // init vars
  uint8_t i;
  for(i=0; i<80; i++){
    cellVoltages[i] = 410 - ((i+3)%6);
    powHistory[i] = 20 - (i%30);
  }
  
  // start display
  u8g2.begin();
  u8g2.clear();
  u8g2.clearBuffer();
  u8g2.clearDisplay();
  delay(100);
    
  // start the CAN bus at 500 kbps
  CAN.setClockFrequency(8e6);
  CAN.setPins(10, 7);
  if (!CAN.begin(500E3)) {
    Serial.println("Starting CAN failed!");
    while (1);
  } else {
    Serial.println("CAN started");
  }

  // register the receive callback
  CAN.onReceive(onReceive);
}

uint8_t x;
uint32_t loopCount = 0;
uint16_t cellMin = 0;
uint16_t b_cellVoltages[80];
float b_powHistory[80];
uint8_t b_powHistoryIndex = 0;
float b_powerKW = 0.4;
float b_cellMinVoltage = 3.71, b_cellMaxVoltage = 4.09;
float b_soc1 = 90, b_soc2 = 89;

void loop() {
  loopCount++;
  
  powHistory[powHistoryIndex] = powerSum/powerNum;
  powerSum = 0;
  powerNum = 0;
  powHistoryIndex++;
  if(powHistoryIndex >= 80) powHistoryIndex = 0;

  // buffering
  b_soc1 = soc1;
  b_soc2 = soc2;
  b_cellMinVoltage = cellMinVoltage;
  b_cellMaxVoltage = cellMaxVoltage;
  b_powerKW = powerKW;
  b_powHistoryIndex = powHistoryIndex;
  for(x=0; x<80; x++){
    b_cellVoltages[x] = cellVoltages[x];
    b_powHistory[x] = powHistory[(x+b_powHistoryIndex)%80];
  }

  drawing = 1;
  u8g2.setFont(u8g2_font_amstrad_cpc_extended_8r);
  u8g2.firstPage();
  do {
    // u8g2.setCursor(-2, 4);
    if (loopCount%2==0) u8g2.drawPixel(0,0);
    
    u8g2.setCursor(8, 9);
    u8g2.print("C-Zero  ");
    u8g2.print(b_soc2, 1);
    u8g2.print("%");

    u8g2.setCursor(85, 28);
    u8g2.print(b_powerKW, 1);
    u8g2.setCursor(85, 37);
    u8g2.print("   KW");

    for(x=0; x<80; x++){
      u8g2.drawLine(x, 32, x, 32-(b_powHistory[x]));
    }

    cellMin = 1000;
    for(x=0; x<80; x++){
      if (b_cellVoltages[x] < cellMin) cellMin =  b_cellVoltages[x];
    }
    cellMin -= 2;
    for(x=0; x<79; x++){
      u8g2.drawLine(x, 64, x, 64-(b_cellVoltages[x]-cellMin));
    }

    u8g2.setCursor(85, 54);
    u8g2.print(b_cellMaxVoltage, 2);
    u8g2.print("v");
    u8g2.setCursor(85, 63);
    u8g2.print(b_cellMinVoltage, 2);
    u8g2.print("v");

  } while ( u8g2.nextPage() );
  drawing = 0;
  
  delay(200);
}
